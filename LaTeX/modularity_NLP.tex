\text{subject to: }\\ 
\sum\limits_{m}^{}(Y_{n,m}) = 1 \forall n\\    
L_{m}  = \sum\limits_{n,e: (n,e) \in LK }^{}(Y_{n,m}Y_{e,m}) \quad \quad \forall m\\
D_{m}  = \sum\limits_{n}^{}(d_{n}Y_{n,m})  \quad \quad \forall m\\
L \in \mathbb R, D \in \mathbb R_+,  Y \in \{0,1\}\\
